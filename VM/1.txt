function updateProtocolVisibility(ports, protocol, protocol_required) {
    var addGroupsValue = parseInt($addGroups, 10);
    var portsValue = $(ports).val();
    var isVisible = (addGroupsValue >= 1 && addGroupsValue <= 3) && !portsValue;

    // Управление видимостью контейнера
    $(protocol_required).toggle(isVisible);

    // Дополнительно: Сброс значения при скрытии
    if (!isVisible) {
        $(protocol).val('').removeClass('invalid');
        $(protocol_required).removeClass('empty');
    }
}

    }

$("#is_ex_change").on("change", function () {
    if (!$(this).hasClass("disabled")) {
        if ($(this).is(":checked")) {
            $("#row_justification").show();
        } else {
            $("#row_justification").hide();
            $("#justification").val(null).change();
            $("#inc_desc").parent().hide();
            $("#inc_desc").val(null).change();
        }
    }
});

$("#justification").on("change", function () {
    if ($("#justification").val() === 'для_выполнения_поручения_руковод') {
        $("#inc_desc").parent().hide();
        $("#inc_desc").val(null).change();
    } else {
        $("#inc_desc").parent().show();
    }
});

if (ITRP.record.initialValues.custom_data["block_ui"] == true) { $(this).hide(); }
$("#correctness_mark").on("change", function () {
    if ($(this).is(":checked")) {
        $("#incorrect_mark").show();
    } else {
        $("#incorrect_mark").hide();
    }
});
$("#is_dis").on("change", function () {
    if ($(this).is(":checked")) {
        $("#correctness_block").show();
    } else {
        $("#correctness_block").hide();
    }
});
